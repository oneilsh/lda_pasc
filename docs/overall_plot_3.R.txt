library(dplyr)
library(ggplot2)
library(scales)
library(stringr)

overall_plot_8 <- function( model_tests_covid_pvals) {
    df <- model_tests_covid_pvals

    unique_topics <- df %>% 
                       pull(topic_name) %>%
                       unique() %>%
                       str_sort(numeric = TRUE)

    topics_subset <- unique_topics[(300-41):300]
    
    do_plot_clean(df, unique_topics, topics_subset, contrast_filter = "Overall", title = "Overall")

    return(NULL)
}
